-- Enable UUID extension if not already enabled
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Function to automatically update updated_at column
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata';
    RETURN NEW;
END;
$$ language 'plpgsql';

-- ============================================
-- 2. CART TABLE (Single entry per user with JSONB)
-- ============================================
CREATE TABLE cart (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    items JSONB NOT NULL DEFAULT '[]'::jsonb,
    -- Format: [
    --   {
    --     "product_id": "uuid",
    --     "product_name": "Classic White Shirt",
    --     "quantity": 2,
    --     "price": 1499.00,
    --     "discounted_price": 1199.00,
    --     "vendor_id": "uuid",
    --     "vendor_name": "Fashion Hub",
    --     "selected_attributes": {"size": "L", "color": "White"},
    --     "applied_coupon": "SAVE10",  -- Coupon applied to this specific product
    --     "discount_amount": 149.90,
    --     "final_price": 1049.10
    --   }
    -- ]
    applied_coupon_code VARCHAR(50), -- Global coupon applied to entire cart
    coupon_discount_amount DECIMAL(10, 2) DEFAULT 0, -- Total discount from global coupon
    subtotal DECIMAL(10, 2) DEFAULT 0,
    total_discount DECIMAL(10, 2) DEFAULT 0,
    grand_total DECIMAL(10, 2) DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata',
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata'
);

CREATE TRIGGER update_cart_updated_at 
BEFORE UPDATE ON cart 
FOR EACH ROW 
EXECUTE FUNCTION update_updated_at_column();

CREATE INDEX idx_cart_user_id ON cart(user_id);

-- ============================================
-- 3. COUPONS AND PROMOCODES TABLE
-- ============================================
CREATE TABLE coupons (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code VARCHAR(50) UNIQUE NOT NULL,
    description TEXT,
    discount_type VARCHAR(20) NOT NULL CHECK (discount_type IN ('PERCENTAGE', 'FIXED_AMOUNT')),
    discount_value DECIMAL(10, 2) NOT NULL CHECK (discount_value >= 0),
    minimum_order_amount DECIMAL(10, 2) DEFAULT 0,
    maximum_discount_amount DECIMAL(10, 2), -- For percentage coupons
    valid_from TIMESTAMP WITH TIME ZONE NOT NULL,
    valid_until TIMESTAMP WITH TIME ZONE NOT NULL,
    usage_limit_per_user INTEGER DEFAULT 1,
    usage_limit_total INTEGER DEFAULT 0, -- 0 means unlimited
    usage_count_total INTEGER DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE,
    -- For product/vendor specific coupons
    applicable_to VARCHAR(20) DEFAULT 'ALL' CHECK (applicable_to IN ('ALL', 'SPECIFIC_VENDOR', 'SPECIFIC_CATEGORY', 'SPECIFIC_PRODUCT')),
    vendor_id UUID REFERENCES vendors(id) ON DELETE CASCADE,
    category_id UUID REFERENCES product_categories(id) ON DELETE CASCADE,
    product_id UUID REFERENCES product_details(id) ON DELETE CASCADE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata',
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata'
);

FOR EACH ROW

CREATE INDEX idx_coupons_valid_date ON coupons(valid_from, valid_until);

-- ============================================
-- 5. USERS PAYMENT AND PURCHASE TABLE (ORDERS)
-- ============================================
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_number VARCHAR(50) UNIQUE NOT NULL,
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    vendor_id UUID REFERENCES vendors(id) ON DELETE SET NULL, -- Multiple vendors per order
    items JSONB NOT NULL, -- Stores products in JSON format
    subtotal DECIMAL(10, 2) NOT NULL DEFAULT 0,
    total_discount DECIMAL(10, 2) DEFAULT 0,
    coupon_discount DECIMAL(10, 2) DEFAULT 0,
    shipping_charge DECIMAL(10, 2) DEFAULT 0,
    tax_amount DECIMAL(10, 2) DEFAULT 0,
    grand_total DECIMAL(10, 2) NOT NULL DEFAULT 0,
    
    -- Shipping details
    shipping_address JSONB NOT NULL,
    delivery_instructions TEXT,
    
    -- Payment details
    payment_method VARCHAR(50) NOT NULL CHECK (payment_method IN ('COD', 'CARD', 'UPI', 'NET_BANKING', 'WALLET')),
    payment_status VARCHAR(50) DEFAULT 'PENDING' CHECK (payment_status IN ('PENDING', 'PAID', 'FAILED', 'REFUNDED')),
    payment_transaction_id VARCHAR(255),
    payment_details JSONB, -- Store payment gateway response
    
    -- Order status
    order_status VARCHAR(50) DEFAULT 'PLACED' CHECK (order_status IN ('PLACED', 'CONFIRMED', 'PROCESSING', 'SHIPPED', 'DELIVERED', 'CANCELLED', 'RETURNED')),
    order_date TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata',
    delivered_at TIMESTAMP WITH TIME ZONE,
    cancelled_at TIMESTAMP WITH TIME ZONE,
    cancellation_reason TEXT,
    
    -- Coupon applied to entire order
    applied_coupon_id UUID REFERENCES coupons(id) ON DELETE SET NULL,
    applied_coupon_code VARCHAR(50),
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata',
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata'
);

FOR EACH ROW 
EXECUTE FUNCTION update_updated_at_column();

-- ============================================
-- 7. VENDOR PAYMENT TABLE (Payments to vendors)
-- ============================================
CREATE TABLE vendor_payments (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vendor_id UUID NOT NULL REFERENCES vendors(id) ON DELETE CASCADE,
    order_id UUID REFERENCES orders(id) ON DELETE SET NULL,
    invoice_number VARCHAR(50) UNIQUE NOT NULL,
    
    -- Payment details
    total_order_amount DECIMAL(10, 2) NOT NULL,
    platform_commission DECIMAL(10, 2) NOT NULL, -- Platform's commission
    commission_percentage DECIMAL(5, 2) NOT NULL, -- Commission percentage
    gst_on_commission DECIMAL(10, 2) DEFAULT 0,
    tds DECIMAL(10, 2) DEFAULT 0, -- Tax Deducted at Source
    payment_amount DECIMAL(10, 2) NOT NULL, -- Final amount to be paid
    payment_note TEXT,
    
    -- Payment status
    payment_status VARCHAR(50) DEFAULT 'PENDING' CHECK (payment_status IN ('PENDING', 'PROCESSING', 'PAID', 'FAILED', 'HOLD')),
    payment_method VARCHAR(50) CHECK (payment_method IN ('BANK_TRANSFER', 'UPI', 'CHEQUE', 'CASH')),
    payment_date TIMESTAMP WITH TIME ZONE,
    transaction_reference VARCHAR(255),
    
    -- Payment cycle
    payment_cycle_from TIMESTAMP WITH TIME ZONE,
    payment_cycle_to TIMESTAMP WITH TIME ZONE,
    
    bank_account_details JSONB, -- Store vendor's bank details for payment
    payment_response JSONB, -- Store payment gateway response if any
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata',
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Kolkata'
);

CREATE TRIGGER update_vendor_payments_updated_at 
BEFORE UPDATE ON vendor_payments 
FOR EACH ROW 
EXECUTE FUNCTION update_updated_at_column();

CREATE INDEX idx_vendor_payments_vendor_id ON vendor_payments(vendor_id);

CREATE INDEX idx_vendor_payments_invoice ON vendor_payments(invoice_number);

-- USEFUL QUERIES
-- ============================================

-- Get user's cart with all items
SELECT 
    u.username,
    c.items,
    c.subtotal,
    c.total_discount,
    c.coupon_discount_amount,
    c.grand_total,
    c.applied_coupon_code
FROM cart c
JOIN users u ON u.id = c.user_id
WHERE c.user_id = 'user_uuid_here' AND c.is_active = true;

-- Get user's order history with product names
SELECT 
    o.order_number,
    o.order_date,
    o.grand_total,
    o.order_status,
    p.product_name,
    p.price
FROM orders o
CROSS JOIN LATERAL jsonb_array_elements(o.items) AS items
JOIN product_details p ON p.id = (items->>'product_id')::uuid
WHERE o.user_id = 'user_uuid_here'
ORDER BY o.order_date DESC;

-- Get vendor payment summary
SELECT 
    v.name,
    COUNT(vp.id) as total_transactions,
    SUM(vp.total_order_amount) as total_order_amount,
    SUM(vp.platform_commission) as total_commission,
    SUM(vp.payment_amount) as total_payable
FROM vendor_payments vp
JOIN vendors v ON v.id = vp.vendor_id
WHERE vp.payment_status = 'PENDING'
GROUP BY v.id, v.name;

-- Get product-wise purchase history
SELECT 
    p.product_name,
    COUNT(ph.id) as total_purchases,
    SUM(ph.purchase_count) as total_quantity_sold,
    SUM(ph.total_spent) as total_revenue
FROM user_product_history ph
JOIN product_details p ON p.id = ph.product_id
GROUP BY p.id, p.product_name
ORDER BY total_revenue DESC;
